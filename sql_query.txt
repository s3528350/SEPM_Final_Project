DROP DATABASE if exists humanoid;
create database humanoid;
use humanoid;
drop table user;
drop table location;
drop table tour;
drop table tour_type;
CREATE TABLE IF NOT EXISTS user (id INT(12) UNSIGNED AUTO_INCREMENT PRIMARY KEY, fname VARCHAR(50) NOT NULL, lname VARCHAR(50) NOT NULL,email VARCHAR(50) NOT NULL, password CHAR(40) NOT NULL, confirmPassword CHAR(40) NOT NULL, rights VARCHAR(40), created_at VARCHAR(40), updated_at VARCHAR(40));
